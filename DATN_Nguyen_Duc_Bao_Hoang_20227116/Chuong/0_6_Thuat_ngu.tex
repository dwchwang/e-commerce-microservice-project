\chapter*{DANH MỤC THUẬT NGỮ}
\phantomsection
\addcontentsline{toc}{chapter}{DANH MỤC THUẬT NGỮ}
\begin{longtable}{p{4cm} p{9.5cm}}
    \hline
    \textbf{Thuật ngữ} & \textbf{Giải nghĩa} \\ \hline
    \textbf{Bounded Context}       & Ranh giới nghiệp vụ trong DDD, bên trong đó mỗi khái niệm có một ý nghĩa nhất quán \\
    \textbf{Circuit Breaker}       & Cơ chế tự động ngắt kết nối khi dịch vụ phía sau liên tục lỗi, tránh lỗi lan truyền dây chuyền \\
    \textbf{Eventual Consistency}  & Mô hình nhất quán cuối cùng: các bản sao dữ liệu sẽ dần hội tụ về cùng một giá trị sau một khoảng thời gian \\
    \textbf{Flash Sale}            & Đợt bán hàng giảm giá với số lượng giới hạn trong thời gian ngắn, tạo lượng truy cập đồng thời rất cao \\
    \textbf{Idempotent Consumer}   & Thành phần đảm bảo xử lý cùng một thông điệp nhiều lần vẫn cho kết quả như xử lý một lần \\
    \textbf{Microservice}          & Dịch vụ nhỏ, độc lập, xây dựng quanh một ranh giới nghiệp vụ, có thể triển khai và mở rộng riêng \\
    \textbf{Outbox Pattern}        & Mẫu thiết kế lưu sự kiện vào bảng \texttt{outbox} chung giao dịch với dữ liệu nghiệp vụ, đảm bảo gửi ít nhất một lần \\
    \textbf{Over-sell}             & Tình trạng bán vượt quá số lượng hàng thực có, do tranh chấp truy cập khi hệ thống tải cao \\
    \textbf{Rate Limiting}         & Kỹ thuật giới hạn số lượng yêu cầu một người dùng được gửi trong một khoảng thời gian \\
    \textbf{Saga Pattern}          & Mẫu thiết kế xử lý giao dịch phân tán bằng chuỗi giao dịch cục bộ; khi lỗi sẽ thực hiện giao dịch bù trừ \\
    \textbf{Service Discovery}     & Cơ chế tự động phát hiện địa chỉ IP:Port của dịch vụ trong môi trường vùng chứa thay đổi liên tục \\
    \textbf{Trusted Subsystem}     & Mẫu thiết kế: chỉ API Gateway kiểm tra JWT, rồi chuyển thông tin người dùng qua header cho backend phía sau \\
    \hline
\end{longtable}
