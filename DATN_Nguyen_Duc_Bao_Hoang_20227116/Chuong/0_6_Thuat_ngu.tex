\chapter*{DANH MỤC THUẬT NGỮ}
\phantomsection
\addcontentsline{toc}{chapter}{DANH MỤC THUẬT NGỮ}

\begin{longtable}{p{4cm} p{9.5cm}}
    \hline
    \textbf{Thuật ngữ} & \textbf{Giải nghĩa} \\ \hline

    \textbf{Bounded Context}       & Ranh giới rõ ràng trong DDD, bên trong đó mọi khái niệm domain có nghĩa nhất quán và duy nhất \\
    \textbf{Circuit Breaker}       & Pattern ngăn cascading failure: tự động ngắt mạch khi service downstream liên tục lỗi \\
    \textbf{Eventual Consistency}  & Mô hình nhất quán trong hệ phân tán: các bản sao dữ liệu sẽ hội tụ về cùng giá trị sau một khoảng thời gian \\
    \textbf{Flash Sale}            & Sự kiện bán hàng số lượng giới hạn trong thời gian ngắn, tạo tải đồng thời cực cao \\
    \textbf{Idempotent Consumer}   & Pattern đảm bảo xử lý cùng một message nhiều lần không gây side effect khác so với xử lý một lần \\
    \textbf{Microservice}          & Service nhỏ, độc lập, xây dựng xung quanh một Bounded Context, có thể triển khai và scale riêng biệt \\
    \textbf{Outbox Pattern}        & Pattern lưu event vào bảng \texttt{outbox} cùng transaction với business data, đảm bảo at-least-once delivery \\
    \textbf{Over-sell}             & Tình trạng bán nhiều hơn số lượng hàng thực có, xảy ra do race condition trong hệ thống tải cao \\
    \textbf{Rate Limiting}         & Kỹ thuật giới hạn số request một client gửi trong một khoảng thời gian nhất định \\
    \textbf{Saga Pattern}          & Pattern xử lý distributed transaction bằng chuỗi local transaction; khi lỗi thực thi compensating transaction \\
    \textbf{Service Discovery}     & Cơ chế tự động phát hiện địa chỉ IP:Port của service trong môi trường container động \\
    \textbf{Trusted Subsystem}     & Pattern: chỉ API Gateway validate JWT, forward thông tin user qua header tin cậy cho backend \\

    \hline
\end{longtable}
